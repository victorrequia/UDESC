    return math.sin((x/(math.sqrt(pow(x,2)+1)))) + 1

def quadratura(f, pontos_e_pesos):
    soma = 0
    for x_k, c_k in pontos_e_pesos:
        soma += c_k * f(x_k)
    return soma
